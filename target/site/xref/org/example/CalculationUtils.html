3   import java.util.Scanner;

5   public class CalculationUtils {
6   
7     private static final String ARITHMETIC_SINGS = "-|/|\\*|\\+";
8   
9     public static double getNumber() {
10      Scanner scanner = new Scanner(System.in);
11      scanner.useDelimiter("\n");
12      System.out.println("Enter the number:");
13      if (scanner.hasNextDouble())
14        return scanner.nextDouble();
15      else {
16        System.out.println("Number is incorrect. Try again");
17        return getNumber();
18      }
19    }
20  
21    public static char getOperation() {
22      System.out.println("Enter the operation (+, -, *, /):");
23      Scanner scanner = new Scanner(System.in);
24      scanner.useDelimiter("\n");
25      char operation;
26      if (scanner.hasNext(ARITHMETIC_SINGS)) {
27        operation = scanner.next().charAt(0);
28      } else {
29        System.out.println("Operation is incorrect. Try again");
30        operation = getOperation();
31      }
32      return operation;
33    }
34  
35    public static double calculation(char operation, double firstNumber, double secondNumber) {
36      Calculator calculate = new Calculator();
37      return switch (operation) {
38        case '+' -> calculate.add(firstNumber, secondNumber);
39        case '-' -> calculate.subtract(firstNumber, secondNumber);
40        case '*' -> calculate.multiply(firstNumber, secondNumber);
41        case '/' -> calculate.divide(firstNumber, secondNumber);
42        default -> 0;
43      };
44    }
45  
46    public static double getResult(double result) {
47      final int CALCULATION_ACCURACY = 2; //A quantity of digits after comma
48      return Math.round((result) * (int) Math.pow(10.0, CALCULATION_ACCURACY)) / Math.pow(10.0,
49          CALCULATION_ACCURACY);
50    }
51  }
